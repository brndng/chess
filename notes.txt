const PrivateRoute = ({ component: Component, ...rest }) => {
  return <Route {...rest} render={(props) => {    
    return auth.isAuthenticated === true 
      ? <Component {...props} />
      : <Redirect to={{
          pathname: '/login',
          state: { from: props.location }
        }} />
  }} />
}
